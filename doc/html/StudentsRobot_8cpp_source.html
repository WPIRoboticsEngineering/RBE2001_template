82 void StudentsRobot::updateStateMachine() {
83  long now = millis();
84  switch (status) {
85  case StartupRobot:
86  //Do this once at startup
87  status = StartRunning;
88  Serial.println("StudentsRobot::updateStateMachine StartupRobot here ");
89  break;
90  case StartRunning:
91  Serial.println("Start Running");
92 
93  digitalWrite(H_BRIDGE_ENABLE, 1);
94  // Start an interpolation of the motors
95  motor1->startInterpolationDegrees(motor1->getAngleDegrees(), 1000, SIN);
96  motor2->startInterpolationDegrees(motor2->getAngleDegrees(), 1000, SIN);
97  motor3->startInterpolationDegrees(motor3->getAngleDegrees(), 1000, SIN);
98  status = WAIT_FOR_MOTORS_TO_FINNISH; // set the state machine to wait for the motors to finish
99  nextStatus = Running; // the next status to move to when the motors finish
100  startTime = now + 1000; // the motors should be done in 1000 ms
101  nextTime = startTime + 1000; // the next timer loop should be 1000ms after the motors stop
102  break;
103  case Running:
104  // Set up a non-blocking 1000 ms delay
105  status = WAIT_FOR_TIME;
106  nextTime = nextTime + 1000; // ensure no timer drift by incremeting the target
107  // After 1000 ms, come back to this state
108  nextStatus = Running;
109 
110  // Do something
111  if (!digitalRead(0))
112  Serial.println(
113  " Running State Machine " + String((now - startTime)));
114  break;
115  case WAIT_FOR_TIME:
116  // Check to see if enough time has elapsed
117  if (nextTime <= millis()) {
118  // if the time is up, move on to the next state
119  status = nextStatus;
120  }
121  break;
122  case WAIT_FOR_MOTORS_TO_FINNISH:
123  if (motor1->isInterpolationDone() && motor2->isInterpolationDone()
124  && motor3->isInterpolationDone()) {
125  status = nextStatus;
126  }
127  break;
128  case Halting:
129  // save state and enter safe mode
130  Serial.println("Halting State machine");
131  digitalWrite(H_BRIDGE_ENABLE, 0);
132  motor3->stop();
133  motor2->stop();
134  motor1->stop();
135 
136  status = Halt;
137  break;
138  case Halt:
139  // in safe mode
