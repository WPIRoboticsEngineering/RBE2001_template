42  if(interpElapsed < duration && duration > 0){
43  // linear elapsed duration
44  unitDuration = interpElapsed / duration;
45  if (mode == SIN) {
46  // sinusoidal ramp up and ramp down

64 void PIDMotor::loop() {
65  if (millis() - lastTimeRunPID > myPID.sampleRateMs) {
66  lastTimeRunPID = millis();
67  } else {
68  return;
69  }
70  // Perform linear/sinusoidal interpolation
71  unitDuration=getInterpolationUnitIncrement();
72  if (unitDuration<1) {
73  float setpointDiff = endSetpoint - startSetpoint;
74  float newSetpoint = startSetpoint + (setpointDiff * unitDuration);
75  Setpoint = newSetpoint;
76  } else {
77  // If there is no interpoation to perform, set the setpoint to the end state
78  Setpoint = endSetpoint;
79  }
80  // Done Performing linear/sinusoidal interpolation
81  calcVel(); // ensure the velocity value is kept up to date
82  if (mode != VEL) {
83  // Position mode
84  Input = (float) getPosition();
85  //check for integral windup
86  bool thisErrPositive = (Setpoint - Input) > 0;
87  if (thisErrPositive != lastErrPositive) {
88  myPID.clearIntegralBuffer();
89  }
90  lastErrPositive = thisErrPositive;
91  // done with integral windup
92  // run the PID compute function
93  Output = myPID.calc(Setpoint, Input);
94  // use the unit vector function to set the output of the PID
95  setOutputUnitVector(Output);
96  } else {
97  velocityLoop();
98  }
99 }

132  // read the stop value from the subclass and set it to the hardware output

214 double PIDMotor::calcVel() {
215  //current positions
216  double curPos = getPosition();
217  //current time
218  curTime = millis();
219  //time change in ms from last call
220  timeInterval = curTime - prevTime;
221  if (timeInterval < 50)
222  return Vel; // If this is polled too fast, return the cached velocity
223  //encoder ticks since last call
224  movement = curPos - prevPos;
225  //encoder ticks to degrees
226  double movementdeg = movement * ticksToDeg();
227  //timeInterval in seconds
228  double timeIntervalsec = timeInterval / 1000.0;
229  //Velocity in degrees per seconds
230  Vel = movementdeg * timeIntervalsec;
231  //Serial.println("Vel "+String(movementdeg)+" Per "+String(timeIntervalsec)+" is "+String(Vel)+" max "+String(getFreeSpinMaxDegreesPerSecond()));
232  //sets curent vals to previous
233  prevPos = curPos;
234  prevTime = curTime;
235  return Vel;
